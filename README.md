class Person {
  constructor(obj) {
    this.name = obj.name;
    this.job_title = obj.job_title;
    this.age = obj.age;
    this.resume = obj.resume;
    this.project = obj.project;
  }

  getCertificate(cert, from) {
    if (!cert) {
      return "no certificate";
    }
    return "get certification from " + from;
  }
}
